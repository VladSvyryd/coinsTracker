@app.route('/users')
def getAllUsers():
    myDB = sqlite3.connect('db/coins.db')

    dbCursor = myDB.cursor()
    dbCursor.execute('SELECT * FROM users')
    print(dbCursor.fetchall())
    return jsonify(dbCursor.fetchall()), 201
